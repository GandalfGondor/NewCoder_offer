/*给定一个数组A[0,1,...,n-1],请构建一个数组B[0,1,...,n-1],其中B中的元素B[i]=A[0]*A[1]*...*A[i-1]*A[i+1]*...*A[n-1]。不能使用除法。*/
class Solution {
public:
    vector<int> multiply(const vector<int>& A) {

	int i = 0;
	int temp = 0;
	int j = 0;
	vector<int> p;
	if (A.size() == 0)
		return p;
	while (i< A.size())
	{
		temp = 1;
		for (j = 0; j<A.size(); j++)
		{
			if (j == i)
				continue;
			temp = temp*A[j];
		}
		i++;
		p.push_back(temp);
	}
	return p;
}
};
