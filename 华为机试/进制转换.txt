#include<iostream>
#include<string>
#include<vector>
#include<math.h>
using namespace std;

int conv(char s1)
{
	switch (s1)
	{
	case 'A':return 10;
	case 'B':return 11;
	case 'C':return 12;
	case 'D':return 13;
	case 'E':return 14;
	case 'F':return 15;
	default:return int(s1-'0');
	}
}
int main()
{
	string s1;
	int i = 0;
	int len = 0;
	int val = 0;
	while (cin >> s1)
	{
		val = 0;
		len = s1.size() - 2;
		for (i = 0; i < len; i++)
		{
			val = val + pow(16, i)*conv(s1[s1.size() - 1 - i]);
		}
		cout << val << endl;
	}
	return 0;
}
