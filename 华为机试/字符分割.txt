#include<iostream>
#include<string>
using namespace std;
void print(string s1)
{
	if (s1 == "")
		return;
	int i = 0, j = 0;
	for (i; i < s1.size() / 8; i++)
	{
		for (j = 0; j < 8; j++)
		{
			cout << s1[i * 8 + j];
		}
		cout << '\n';
	}
	for (j = 0; i * 8 + j < s1.size(); j++)
	{
		cout << s1[i * 8 + j];
	}
	while (j!=0&&j++<8)
		cout << 0;

}
int main()
{
	string s1, s2;
	getline(cin, s1);
	getline(cin, s2);
	print(s1);	cout << '\n';
	print(s2);
	return 0;
}
