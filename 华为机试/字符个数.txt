#include<iostream>
using namespace std;
int main()
{
	char des; char des2;
	char *a = new char[5000];
	int j = 0;
	cin.getline(a, 5000);
	cin >> des;
	if (des <= 90) des2 = des + 32;
	else des2 = des - 32;
	int cnt = 0;
	while (a[j] != '\0')
	{
		if (a[j] == des || a[j] == des2) cnt++;
		j++;
	}
	cout << cnt;
	return 0;
}
