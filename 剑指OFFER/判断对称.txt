class Solution {
public:
bool iscore(TreeNode* pl,TreeNode* pr)
{
	if(pl == NULL && pr == NULL)
		return true;
    if(pl == NULL || pr == NULL)
		return false;
	if(pl->val == pr->val)
		return iscore(pl->right,pr->left) && iscore(pl->left,pr->right);
	else return false;
}

bool isSymmetrical(TreeNode* pRoot)
{
	if(pRoot == NULL)
		return true;
	return iscore(pRoot->left,pRoot->right);
}

};
