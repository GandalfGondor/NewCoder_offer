/* 在一个二维数组中，每一行都按照从左到右递增的顺序排序，
每一列都按照从上到下递增的顺序排序。请完成一个函数，输入
这样的一个二维数组和一个整数，判断数组中是否含有该整数。 */
class Solution {
public:
    bool Find(int target, vector<vector<int> > array) {
        int len = array.size();
        int wid = array[0].size();
        int i,j;
        bool flag = false;
        for(i=0;i<len;i++)
            for(j=0;j<wid;j++)
            {
                if(array[i][wid-1]<target)
                    continue;
                if(array[i][j]==target)
                {   flag = true;
                    break;
                }
            }
        return flag;
    }
};
