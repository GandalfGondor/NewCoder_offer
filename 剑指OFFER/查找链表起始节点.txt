/*
一个链表中包含环，请找出该链表的环的入口结点。

struct ListNode {
    int val;
    struct ListNode *next;
    ListNode(int x) :
        val(x), next(NULL) {
    }
};
*/
class Solution {
public:
 ListNode* EntryNodeOfLoop(ListNode* pHead)
{
	vector<ListNode*> v;
	v.push_back(pHead);
	ListNode* p = pHead->next;
	ListNode* temp;
	int i = 0;
	for (;p != nullptr;p = p->next)
	{
		for (i = 0; i < v.size(); i++)
		{
			if (v[i] == p)
			{
				return p;
			}
		}
		v.push_back(p);
	}
	return nullptr;

}
};
