大家都知道斐波那契数列，现在要求输入一个整数n，请你输出斐波那契数列的第n项。
n<=39


class Solution {
public:
    int Fibonacci(int n) {
        if(n==0) return 0;
        int one = 1;
        int two = 1;
        int three = 2;
        int i =3;
        if(n ==1||n==2)
            return 1;
        while(i<n)
        {
            one = two;
            two = three;
            three = one + two;
            i++;
        }
        return three;
    
    }
};
