int func(int num,vector<int> coin) 
{
	vector<vector<int>> dp(num+1,vector<int>(coin.size(),0));
	int j=0,i = 0;
	for(i=1;i<=num;++i)
	{
		for(j=0;j<coin.size();++j)
		{
			if(coin[j] <= i)
				dp[i][j] = min(dp[i-1][j],dp[i-coin[j]][j])+1;
			else dp[i][j] = dp[i-1][j]+1;
		}
	}
	int res = dp.back().back();
	for(auto i :dp.back())
	{
		if(res>i)
			res = i;
	}
	return res;
}
