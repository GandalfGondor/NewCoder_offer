/* 请实现一个函数，将一个字符串中的空格替换成“%20”。例如，当字符串为We Are Happy.则经过替换之后的字符串为We%20Are%20Happy。*/
class Solution {
public:
	void replaceSpace(char *str,int length) {
    char * temp = new char[length];
		int i;
		int j = 0;

		for (i = 0; i<length+1; i++)
		{
			if (str[i] != ' ')
			{
				temp[j] = str[i];
				j++;
			}
			else
			{
				temp[j] = '%';
				temp[j + 1] = '2';
				temp[j + 2] = '0';
				j = j + 3;
			}
		}
		for (i = 0; i < j+1; i++)
		{
			str[i] = temp[i];
		}
	}
};
